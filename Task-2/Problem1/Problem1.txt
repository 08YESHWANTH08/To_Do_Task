This Program will given 'True' if a given number is divisible by 3 and 'False' if a given number is not divisible by 3.

FUNCTION CHECKER(Integer num) 
    sum <- 0
    WHILE num>0
        sum <- sum+(num%10)
        num <- num/10
    IF sum>9 THEN
        CALL FUNCTION CHECKER(num)
    ELSE
        IF sum==0 THEN
            OUTPUT "True"
        ELSE
            IF sum==3 THEN
                OUTPUT "True"
            ELSE
                IF sum==6 THEN
                    OUTPUT "True"
                ELSE
                    IF sum==9 THEN
                        OUTPUT "True"
                    ELSE
                        OUTPUT "False"
                    ENDIF
                ENDIF
            ENDIF
        ENDIF
    ENDIF
ENDFUNCTION
        

INPUT num #Enter A Number To Check Whether It Is Divisible By 3 or not
IF num<9 THEN
    IF num==0 THEN
        OUTPUT "True"
    ELSE
        IF num==3 THEN
            OUTPUT "True"
        ELSE
            IF num==6 THEN
                OUTPUT "True"
            ELSE
                IF num==9 THEN
                    OUTPUT "True"
                ELSE
                    OUTPUT "False"
                ENDIF
            ENDIF
        ENDIF
    ENDIF
ELSE
    CALL FUNCTION CHECKER(num)
